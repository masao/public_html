#!/opt/free/bin/perl -w
# -*- CPerl -*-
# $Id$

use strict;

main();
sub main {
    open(PASSWD, "/etc/passwd") || die "open fail: $!";
    my @tmp = <PASSWD>;
    chomp @tmp;
    close(PASSWD);

    srand;
    my $user = undef;
    while (1) {
	my @entry = split(/:/, $tmp[int(rand($#tmp))]);
	$user = $entry[0];
	last if ($entry[6] eq "1");
    }
    print "Location: http://www.ulis.ac.jp/~$user/\n\n";
}
